CREATE VIEW public.v_reservas (
   id,
   reser_id_persona,
   reser_id_discoteca,
   reser_dtm_fecha_reserva,
   reser_str_hora,
   reser_num_cantidad_personas,
   reser_str_motivo,
   reser_servicio_decoracion,
   reser_id_reserva,
   reser_str_estado,
   per_str_primer_nombre,
   per_str_segundo_nombre,
   per_str_primer_apellido,
   per_str_segundo_apellido,
   per_str_email,
   disco_str_nombre,
   disco_str_direccion,
   disco_str_barrio,
   disco_str_num_telefono,
   disco_str_num_celular,
   disco_str_url_ubicacion,
   disco_str_telefono_reserva)
AS
SELECT r.id,
   r.id_persona AS reser_id_persona,
   r.id_discoteca AS reser_id_discoteca,
   r.dtm_fecha_reserva AS reser_dtm_fecha_reserva,
   r.str_hora AS reser_str_hora,
   r.num_cantidad_personas AS reser_num_cantidad_personas,
   r.str_motivo AS reser_str_motivo,
   r.servicio_decoracion AS reser_servicio_decoracion,
   r.id_reserva AS reser_id_reserva,
   r.str_estado AS reser_str_estado,
   p.str_primer_nombre AS per_str_primer_nombre,
   p.str_segundo_nombre AS per_str_segundo_nombre,
   p.str_primer_apellido AS per_str_primer_apellido,
   p.str_segundo_apellido AS per_str_segundo_apellido,
   p.str_email AS per_str_email,
   d.str_nombre AS disco_str_nombre,
   d.str_direccion AS disco_str_direccion,
   d.str_barrio AS disco_str_barrio,
   d.str_num_telefono AS disco_str_num_telefono,
   d.str_num_celular AS disco_str_num_celular,
   d.str_url_ubicacion AS disco_str_url_ubicacion,
   d.str_telefono_reserva AS disco_str_telefono_reserva
FROM tbl_reservas r,
   tbl_personas p,
   tbl_discotecas d
WHERE r.id_persona::text = p.str_num_identificacion::text AND
   r.id_discoteca::text = d.str_identificacion::text;
