CREATE TABLE public.tbl_discotecas (
 id SERIAL,
 str_identificacion VARCHAR(15) NOT NULL,
 str_nombre VARCHAR(500) NOT NULL,
 str_direccion VARCHAR(500),
 str_barrio VARCHAR(255),
 str_num_telefono VARCHAR(10),
 str_num_celular VARCHAR(15) NOT NULL,
 str_url_ubicacion VARCHAR NOT NULL,
 str_horarios VARCHAR NOT NULL,
 str_telefono_reserva VARCHAR(100) NOT NULL,
 str_rango_precios VARCHAR,
 num_estrellas INTEGER DEFAULT 0 NOT NULL,
 str_estado VARCHAR(1) DEFAULT 'A'::character varying NOT NULL,
 id_vestuario VARCHAR(10) NOT NULL,
 id_zona VARCHAR(10) NOT NULL,
 id_genero VARCHAR(10) NOT NULL,
 id_tipo_rumba VARCHAR(10) NOT NULL,
 "createdAt" TIMESTAMP WITH TIME ZONE DEFAULT now() NOT NULL,
 "updatedAt" TIMESTAMP WITH TIME ZONE DEFAULT now() NOT NULL,
 num_cupo_reservas INTEGER DEFAULT 1 NOT NULL,
 str_geolocalizacion VARCHAR,
 str_facebook VARCHAR(1),
 str_logo VARCHAR,
 CONSTRAINT tbl_discotecas_pkey PRIMARY KEY(id),
 CONSTRAINT tbl_discotecas_str_identificacion_key UNIQUE(str_identificacion),
 CONSTRAINT tbl_discotecas_genero_fk FOREIGN KEY (id_genero)
   REFERENCES public.tbl_parametros(str_valor)
   ON DELETE RESTRICT
   ON UPDATE CASCADE
   NOT DEFERRABLE,
 CONSTRAINT tbl_discotecas_tio_rumba_fk FOREIGN KEY (id_tipo_rumba)
   REFERENCES public.tbl_parametros(str_valor)
   ON DELETE RESTRICT
   ON UPDATE CASCADE
   NOT DEFERRABLE,
 CONSTRAINT tbl_discotecas_vestuario_fk FOREIGN KEY (id_vestuario)
   REFERENCES public.tbl_parametros(str_valor)
   ON DELETE RESTRICT
   ON UPDATE CASCADE
   NOT DEFERRABLE,
 CONSTRAINT tbl_discotecas_zona_fk FOREIGN KEY (id_zona)
   REFERENCES public.tbl_parametros(str_valor)
   ON DELETE RESTRICT
   ON UPDATE CASCADE
   NOT DEFERRABLE
)
