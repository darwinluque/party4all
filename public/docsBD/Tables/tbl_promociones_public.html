CREATE TABLE public.tbl_promociones (

 str_url_imagen VARCHAR(255),
 dtm_fecha_promocion TIMESTAMP WITH TIME ZONE NOT NULL,

 "updatedAt" TIMESTAMP WITH TIME ZONE DEFAULT now() NOT NULL,
