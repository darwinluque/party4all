CREATE TABLE public.tbl_personas (
 id SERIAL,
 id_genero_sexo VARCHAR(10),
 id_estado_civil VARCHAR(10),
 id_bebida VARCHAR(10),
 num_estrellas INTEGER DEFAULT 0 NOT NULL,
 str_primer_nombre VARCHAR(50) NOT NULL,
 str_segundo_nombre VARCHAR(50),
 str_primer_apellido VARCHAR(50),
 str_segundo_apellido VARCHAR(50),
 str_tipo_id VARCHAR(10) NOT NULL,
 str_num_identificacion VARCHAR(15) NOT NULL,
 str_email VARCHAR(255) NOT NULL,
 str_token VARCHAR,
 str_condiciones VARCHAR NOT NULL,
 str_acepta_tart VARCHAR(1) DEFAULT 'S'::character varying NOT NULL,
 dtm_fecha_acepta_trat TIMESTAMP WITH TIME ZONE DEFAULT now() NOT NULL,
 "createdAt" TIMESTAMP WITH TIME ZONE DEFAULT now() NOT NULL,
 "updatedAt" TIMESTAMP WITH TIME ZONE DEFAULT now() NOT NULL,
 dtm_fecha_nacimiento DATE,
 str_password VARCHAR DEFAULT 'passw0rd'::character varying NOT NULL,
 str_celular VARCHAR(15),
 str_facebook_id VARCHAR,
 CONSTRAINT tbl_personas_idx UNIQUE(str_num_identificacion),
 CONSTRAINT tbl_personas_pkey PRIMARY KEY(id),
 CONSTRAINT tbl_personas_str_email_key UNIQUE(str_email),
 CONSTRAINT tbl_personas_bebida_fk FOREIGN KEY (id_bebida)
   REFERENCES public.tbl_parametros(str_valor)
   ON DELETE RESTRICT
   ON UPDATE CASCADE
   NOT DEFERRABLE,
 CONSTRAINT tbl_personas_est_civil_fk FOREIGN KEY (id_estado_civil)
   REFERENCES public.tbl_parametros(str_valor)
   ON DELETE RESTRICT
   ON UPDATE CASCADE
   NOT DEFERRABLE,
 CONSTRAINT tbl_personas_sexo_fk FOREIGN KEY (id_genero_sexo)
   REFERENCES public.tbl_parametros(str_valor)
   ON DELETE RESTRICT
   ON UPDATE CASCADE
   NOT DEFERRABLE,
 CONSTRAINT tbl_personas_tipo_id_fk FOREIGN KEY (str_tipo_id)
   REFERENCES public.tbl_parametros(str_valor)
   ON DELETE RESTRICT
   ON UPDATE CASCADE
   NOT DEFERRABLE
)
WITH (oids = false);
